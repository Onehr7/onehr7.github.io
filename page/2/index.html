Python装饰器实现日志输出 2020-11-20|python
# -*- coding: utf-8 -*-

from functools import wraps
import logging


def _create_logger():
    logger = logging.getLogger('LogError')
    logger.setLevel(logging.ERROR)
    file_handler = logging.FileHandler(r'./MJlogs.log')

    log_format = '%(levelname)s %(asctime)s %(message)s'
    formatter = logging.Formatter(log_format)
    file_handler.setFormatter(formatter)

    logger.addHandler(file_handler)
    return logger


def logged(func):
    @wraps(func)
    def do_logging(*args, **kwargs):
        try:
            return func(*args, **kwargs)
        except Exception as e:
            logger = _create_logger()
            error_msg = 'And error has occurred at /' + func.__name__ + '\n'
            logger.exception(error_msg)
            return e
    return do_logging

Python生产者消费者模型 2020-11-06|python|并发
import time
import queue
import threading


class ThreadProductorConsumer():

    # 初始化模型
    def __init__(self):
        # 大小为15的缓冲池，用于容纳产品
        self.q = queue.Queue(15)
        self.screen_lock = threading.Semaphore(value=1)

    def productor(self, product):
        # 生产者不停的每3秒生产一个产品
        while True:
            self.q.put(product)
            cur_time = time.strftime('%Y-%m-%d %H:%M:%S', time.localtime(time.time()))
            cur_name = threading.current_thread().name
            # print是线程不安全的,需要信号量确保没有冲突
            self.screen_lock.acquire()
            print(cur_time+"  " + cur_name + " 生产了一个产品", sep='\n')
            self.screen_lock.release()
            time.sleep(3)

    def consumer(self):
        # 消费者不停的每2秒消费一个产品
        while True:
            self.q.get()
            cur_time = time.strftime('%Y-%m-%d %H:%M:%S', time.localtime(time.time()))
            cur_name = threading.current_thread().name
            self.screen_lock.acquire()
            print(cur_time + "  " + cur_name + " 消费了一个产品", sep='\n')
            self.screen_lock.release()
            time.sleep(2)


if __name__ == "__main__":
    tpc = ThreadProductorConsumer()

    p = "product"
    # 实例化了3个生产者
    for i in range(3):
        t = threading.Thread(target=tpc.productor, args=(p,))
        t.start()

    # 实例化了6个消费者
    for j in range(6):
        v = threading.Thread(target=tpc.consumer)
        v.start()

asyncio.Semaphore控制协程并发量 2020-10-14|python|协程
# -*- coding: utf-8 -*-
# @Time    : 2020/10/12 12:03
# @Author  : Wanghairui

import asyncio
import time


async def main():
    # 控制并发数 20
    sem = asyncio.Semaphore(20)
    tasks = [asyncio.create_task(print_something("c"+str(i), sem)) for i in range(60)]
    start = time.perf_counter()
    await asyncio.gather(*tasks)
    print("运行结束")
    res = [t.result() for t in tasks]
    print(res)
    elapsed = (time.perf_counter() - start)
    print(elapsed)


async def print_something(thing, sem):
    async with sem:
        print(thing)
        await asyncio.sleep(3)
    return "A"


if __name__ == "__main__":
    asyncio.run(main())

归并排序 (Merge)

实现原理

  • 通过分而治之的思想，将两个两个的有序子列进行归并
  • 时间复杂度：稳定在nlogn
  • 优点：稳定的算法
  • 缺点：比较占用内存

实现代码

            System.out.print(n + " ");
        &#125;

    public void merge(int[] a, int left, int mid, int right, int[] tmp)&#123;
        int le = left;
        int m = mid + 1;
        int i = 0;
        while (le <= mid && m <= right)&#123;
            if (a[le] < a[m])&#123;
                tmp[i++] = a[le++];
            &#125;
            else&#123;
                tmp[i++] = a[m++];
            &#125;
        &#125;

            tmp[i++] = a[le++];
        &#125;

并发编程基础二 2019-11-06|Java并发|并发

Java并发编程基础二

1、多线程并发编程

  • 并发：同一时间段内多个任务同时执行，且执行没有结束
  • 并行：单位时间内多个任务同时执行

2、线程安全问题

  • 只读不会出现问题，至少有一个线程进行修改
  • 通常使用同步锁synchronized进行上锁

3、共享变量的内存可见性问题

  • 变量放在主内存中，线程使用变量时，会将主内存中变量复制到自己的工作空间，然后操作自己工作空间中的变量
  • 通过volatile解决

4、synchronized关键字

  • 当一个线程获取到这个锁后，其他线程必须等待该线程释放锁后才能获取该锁
  • 使用synchronized就是将要使用的变量从工作内存中清除，直接从主存中获取变量
  • 退出synchronized就是将本地内存修改的变量刷新到主存中
  • 使用synchronized会导致上下文切换，增加调度开销
    public synchronized void set（int value）&#123;&#125;

5、volatile关键字

  • 确保对一个变量的更新对其他线程马上可见
  • 从主内存中获取，修改后再刷新回主内存
  • 不保证操作的原子性
  • 适合的使用条件
    • 写入变量值不依赖变量当前值
private volatile int value；

6、原子性操作

  • 含义：执行一系列操作时，这些操作要么全执行，要么全不执行
  • 保证原子性：通过synchronized关键字

7、CAS操作

  • 锁的缺点：当线程没有获取到锁被阻塞时，会导致线程上下文切换和重新调度的开销
  • CAS：Compare and Swap，通过硬件保证原子性
  • boolean compareAndSwapLong（Object obj，long valueOffset，long expect，long update），参数为：对象内存位置、对象中变量的偏移量、变量预期值、新的值。如果对象obj中内存偏移量为valueOffset的变量值为expect，则使用新的值update替换expect
ABA问题
- 线程1准备用CAS将变量的值由A替换为B，在此之前，线程2将变量的值由A替换为C，又由C替换为A，然后线程1执行CAS时发现变量的值仍然为A，所以CAS成功。但实际上这时的现场已经和最初不同了，尽管CAS成功，但可能存在潜藏的问题
- 变量的状态值产生了环形转换     A->B,B->A
- 避免：
    + 1、变量的值只能朝一个方向转换    A->B,B->C
    + 2、JDK中AtomicStampedReference类给每个变量状态值配备了时间戳
