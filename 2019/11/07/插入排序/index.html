插入排序

实现原理

  • 从序列中不断取出元素，插入已排序的序列中
  • 时间复杂度：最好O(N),最坏O(N^2)

实现代码

/**
 * @author Mr.wang
 * @date 2019-11-07 15:07
 * function_:
 */
public class InsertSort &#123;
    public static void insertSort(int [] list)&#123;
        for(int i = 1;i < list.length; i++)&#123;
            int tmp = list[i];
            int k;

            //将比现在大的元素向后移
            for(k = i; k > 0 && list[k-1] > tmp; k--)&#123;
                list[k] = list[k - 1];
            &#125;
            list[k] = tmp;
        &#125;
    &#125;
    public static void main(String[] args) &#123;
        int [] list = &#123;29,10,14,37,14&#125;;
        insertSort(list);
        for(int a : list)&#123;
            System.out.print(a+" ");
        &#125;
    &#125;
