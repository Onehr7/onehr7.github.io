归并排序 (Merge)

实现原理

  • 通过分而治之的思想，将两个两个的有序子列进行归并
  • 时间复杂度：稳定在nlogn
  • 优点：稳定的算法
  • 缺点：比较占用内存

实现代码

    public static void main(String[] args)&#123;

        int[] a = &#123;11,44,23,67,88,65,34,48,9,12&#125;;

    public void merge(int[] a, int left, int mid, int right, int[] tmp)&#123;
        int le = left;
        int m = mid + 1;
        int i = 0;
        while (le <= mid && m <= right)&#123;
            if (a[le] < a[m])&#123;
                tmp[i++] = a[le++];
            &#125;
            else&#123;
                tmp[i++] = a[m++];
            &#125;
        &#125;

        //当左右两边仍有剩余时
        while (le <= mid)&#123;
            tmp[i++] = a[le++];
        &#125;
        while (m <= right)&#123;
            tmp[i++] = a[m++];
        &#125;

        //复制回原数组
        for (int j = 0; j < i; j++)&#123;
            a[left+j] = tmp[j];
        &#125;
    &#125;

    public void mergeSort(int[] a,int start, int end,int[] tmp)&#123;
        if(start < end)&#123;
            int mid = (start + end) / 2;
            mergeSort(a, start, mid, tmp);
            mergeSort(a, mid + 1, end, tmp);
            merge(a, start, mid, end,tmp);
        &#125;
        System.out.print("");
