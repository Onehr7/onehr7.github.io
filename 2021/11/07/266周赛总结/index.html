  1. 统计字符串中的元音子字符串

    暴力，两次遍历找出符合条件的数目

    class Solution:
    def countVowelSubstrings(self, word: str) -> int:
        res = 0

        word_set = set()

        for i in range(len(word)):
            if word[i] in 'aeiou':
                for j in range(i, len(word)):
                    if word[j] in ['a', 'e', 'i', 'o', 'u']:
                        word_set.add(word[j])
                    else:
                        word_set = set()
                        break

                    if 'a' in word_set and 'e' in word_set and 'i' in word_set and 'o' in word_set and 'u' in word_set:
                        res += 1
                word_set = set()
            else:
                continue
        return res

  1. 所有子字符串中的元音

    双指针，

    class Solution:
    def countVowels(self, word: str) -> int:
        left = right = 0
        res = 0
        cur = 0
        pre = 0
        while right < len(word):
            if word[right] in 'aeiou':
                cur += 1
            pre += cur
            right += 1

        pw = word[0]
        res += pre

        left += 1
        while left < len(word):
            if pw in 'aeiou':
                pre = pre - (len(word) - left + 1)
            res += pre
            pw = word[left]
            left += 1

        return res
