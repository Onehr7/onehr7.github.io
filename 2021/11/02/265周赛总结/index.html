    # Definition for singly-linked list.
# class ListNode:
#     def __init__(self, val=0, next=None):
#         self.val = val
#         self.next = next
class Solution:
    def nodesBetweenCriticalPoints(self, head: Optional[ListNode]) -> List[int]:
        point_list = []
        if not head:
            return [-1, -1]
        i = 0
        res = []
        while head.next:
            pre = head.val
            head = head.next
            i += 1
            if head.next and pre < head.val > head.next.val:
                point_list.append(i)
            if head.next and pre > head.val < head.next.val:
                point_list.append(i)

        if len(point_list) < 2:
            return [-1, -1]

        res = [point_list[1] - point_list[0], point_list[-1] - point_list[0]]

        for i in range(1, len(point_list)):
            res[0] = min(res[0], point_list[i] - point_list[i-1])

            if point_list[i] - point_list[i-1] == 1:
                break
        return res

    class Solution:
    def minimumOperations(self, nums: List[int], start: int, goal: int) -> int:
        q = deque([start])
        visited = set()
        step = 0

        while q:
            size = len(q)
            while size:
                cur = q.popleft()
                for num in nums:
                    cal = [num + cur, cur - num, cur ^ num]
                    for x in cal:
                        if x == goal:
                            return step + 1
                        if 0 <= x <= 1000 and x not in visited:
                            visited.add(x)
                            q.append(x)
                size -= 1
            step += 1
        return -1
