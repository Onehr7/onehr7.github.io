  1. 值相等的最小索引

    模拟遍历，满足条件则返回

    class Solution:
    def smallestEqual(self, nums: List[int]) -> int:
        for i in range(len(nums)):
            if i % 10 == nums[i]:
                return i
        return -1

  1. 找出临界点之间的最小和最大距离

    遍历得出所有的极大极小值点列表

    遍历极大极小值列表，找出最大最小距离

    # Definition for singly-linked list.
# class ListNode:
#     def __init__(self, val=0, next=None):
#         self.val = val
#         self.next = next
class Solution:
    def nodesBetweenCriticalPoints(self, head: Optional[ListNode]) -> List[int]:
        point_list = []
        if not head:
            return [-1, -1]
        i = 0
        res = []
        while head.next:
            pre = head.val
            head = head.next
            i += 1
            if head.next and pre < head.val > head.next.val:
                point_list.append(i)
            if head.next and pre > head.val < head.next.val:
                point_list.append(i)

        if len(point_list) < 2:
            return [-1, -1]

        res = [point_list[1] - point_list[0], point_list[-1] - point_list[0]]

        for i in range(1, len(point_list)):
            res[0] = min(res[0], point_list[i] - point_list[i-1])

            if point_list[i] - point_list[i-1] == 1:
                break
        return res

  2. [转化数字的最小运算数

    典型无权最短路，使用BFS求解

    😭开始想成了回溯

    class Solution:
    def minimumOperations(self, nums: List[int], start: int, goal: int) -> int:
        q = deque([start])
        visited = set()
        step = 0

        while q:
            size = len(q)
            while size:
                cur = q.popleft()
                for num in nums:
                    cal = [num + cur, cur - num, cur ^ num]
                    for x in cal:
                        if x == goal:
                            return step + 1
                        if 0 <= x <= 1000 and x not in visited:
                            visited.add(x)
                            q.append(x)
                size -= 1
            step += 1
        return -1
